// ── Quick Pay ──────────────────────────────────────────────────── async function doQuickPay() { const mobile = document.getElementById('pay-mobile').value.trim(); const amt = document.getElementById('pay-amount').value.trim(); const remark = document.getElementById('pay-remark').value.trim(); if (!mobile || mobile.length !== 10) return toast('10-digit mobile daalo', 'error'); if (!amt) return toast('Amount daalo', 'error'); // Find circle entry for this mobile const btn = document.getElementById('quick-pay-btn'); btnLoad(btn, true); try { // Get my circle and find matching member const r = await fetch(`${API}/circle/my-circle`, { headers: { 'Authorization': `Bearer ${token}` } }); const d = await r.json(); const match = d.data?.find(c => c.member_id?.mobile === mobile && c.status === 'active'); if (!match) { btnLoad(btn, false); return toast('Ye user aapke active circle mein nahi hai. Pehle add karo.', 'error'); } const r2 = await fetch(`${API}/circle/pay`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` }, body: JSON.stringify({ circle_id: match._id, amount: amt, remark }) }); const d2 = await r2.json(); if (d2.status === 'success') { document.getElementById('pay-mobile').value = ''; document.getElementById('pay-amount').value = ''; document.getElementById('pay-remark').value = ''; document.getElementById('pay-chip').classList.remove('show'); showReceipt(d2.receipt); } else toast(d2.message || 'Payment failed', 'error'); } catch(e) { toast('Network error', 'error'); } finally { btnLoad(btn, false); } } // ── Receipt ────────────────────────────────────────────────────── function showReceipt(rec) { const isPending = rec.mode === 'approve' && rec.status === 'pending'; document.getElementById('receipt-icon').innerHTML = isPending ? '⏳' : '✓'; document.getElementById('receipt-icon').style.background = isPending ? 'linear-gradient(135deg,var(--orange),#f59e0b)' : 'linear-gradient(135deg,var(--green),#22c55e)'; document.getElementById('receipt-title').textContent = isPending ? 'Approval Requested' : 'Payment Successful!'; document.getElementById('r-amount').textContent = `₹${parseFloat(rec.amount).toFixed(2)}`; document.getElementById('r-to').textContent = rec.to_name || '—'; document.getElementById('r-from').textContent = rec.from_name || '—'; document.getElementById('r-remark').textContent = rec.remark || '—'; document.getElementById('r-txid').textContent = rec.tx_id; document.getElementById('r-date').textContent = rec.timestamp; document.getElementById('r-status').textContent = isPending ? 'Pending Approval' : 'Success'; const statusRow = document.getElementById('r-status-row'); statusRow.className = isPending ? 'receipt-row r-pending' : 'receipt-row'; document.getElementById('receipt-overlay').classList.add('open'); } function closeReceipt() { document.getElementById('receipt-overlay').classList.remove('open'); } // ── Helpers ────────────────────────────────────────────────────── function btnLoad(btn, on) { btn.classList.toggle('loading', on); btn.disabled = on; } function toast(msg, type = '') { const t = document.getElementById('toast'); t.textContent = msg; t.className = `toast show ${type}`; setTimeout(() => t.className = 'toast', 3200); } // Init loadCircle();
